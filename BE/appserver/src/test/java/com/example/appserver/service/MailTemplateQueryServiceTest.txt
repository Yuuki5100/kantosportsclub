package com.example.appserver.service;

import com.example.servercommon.entity.MailTemplateEntity;
import com.example.servercommon.repository.MailTemplateRepository;
import com.example.servercommon.responseModel.TemplateSummaryResponse;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.Mockito;

import java.util.List;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

class MailTemplateQueryServiceTest {

    private MailTemplateRepository repository;
    private MailTemplateQueryService service;

    @BeforeEach
    void setUp() {
        repository = mock(MailTemplateRepository.class);
        service = new MailTemplateQueryService(repository);
    }

    @Test
    void getAllSummaries_ShouldReturnMappedList() {
        // モックデータ作成
        MailTemplateEntity entity1 = new MailTemplateEntity();
        entity1.setTemplateName("Template1");
        entity1.setLocale("ja");

        MailTemplateEntity entity2 = new MailTemplateEntity();
        entity2.setTemplateName("Template2");
        entity2.setLocale("en");

        when(repository.findAll()).thenReturn(List.of(entity1, entity2));

        // 実行
        List<TemplateSummaryResponse> result = service.getAllSummaries();

        // 検証
        assertNotNull(result);
        assertEquals(2, result.size());

        assertEquals("Template1", result.get(0).getTemplateName());
        assertEquals("ja", result.get(0).getLocale());

        assertEquals("Template2", result.get(1).getTemplateName());
        assertEquals("en", result.get(1).getLocale());

        verify(repository, times(1)).findAll();
    }

    @Test
    void getAllSummaries_ShouldReturnEmptyList_WhenRepositoryEmpty() {
        when(repository.findAll()).thenReturn(List.of());

        List<TemplateSummaryResponse> result = service.getAllSummaries();

        assertNotNull(result);
        assertTrue(result.isEmpty());

        verify(repository, times(1)).findAll();
    }
}
